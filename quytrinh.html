Quy Trình Tích Hợp Cơ Sở Dữ Liệu Vào JavaFX

Tài liệu hướng dẫn từng bước áp dụng mô hình MVC và DAO để quản lý ấn phẩm thư viện bằng Java, JavaFX và MySQL.

Tổng Quan Kiến Trúc (MVC + DAO)

Giải thích đơn giản: Thay vì nhồi nhét tất cả code vào một file, chúng ta chia ứng dụng thành các lớp (layer) riêng biệt. Giống như một nhà hàng: Giao diện (View) là phòng ăn, Controller là người bồi bàn, Model là món ăn, DAO là đầu bếp, và Database là kho nguyên liệu.

Giải thích kỹ thuật: Áp dụng Separation of Concerns. Controller chỉ xử lý sự kiện UI, gọi các phương thức từ DAO. DAO chứa toàn bộ logic truy xuất dữ liệu (JDBC). Model đại diện cho thực thể kinh doanh (POJO).

  • View (.fxml): Chứa giao diện nút bấm, bảng biểu.
  • Controller (.java): Nhận thao tác từ View, kiểm tra tính hợp lệ, gọi DAO.
  • DAO (.java): Chứa câu lệnh SQL, thực thi truy vấn đến Database.
  • Model (.java): Các lớp đối tượng như AnPham, Sach, TapChi.

I. Thiết Kế Cơ Sở Dữ Liệu

Sử dụng MySQL. Trong thực tế, bạn có thể thiết kế một bảng `an_pham` dùng chung và có cột `loai_an_pham` để phân biệt, hoặc tạo các bảng riêng biệt. Dưới đây là ví dụ tạo bảng `sach` cơ bản.

-- Tạo cơ sở dữ liệu
CREATE DATABASE quanlythuvien;
USE quanlythuvien;

-- Tạo bảng Sách
CREATE TABLE sach (
    ma_sach VARCHAR(20) PRIMARY KEY,
    ten_sach VARCHAR(255) NOT NULL,
    nha_xuat_ban VARCHAR(100),
    so_luong INT DEFAULT 0
);

-- Thêm dữ liệu mẫu
INSERT INTO sach (ma_sach, ten_sach, nha_xuat_ban, so_luong) 
VALUES ('S001', 'Lập trình Java Cơ bản', 'NXB Giáo Dục', 10);

II. Cấu Hình Kết Nối (DatabaseHelper)

Lớp này chịu trách nhiệm duy nhất là tạo và trả về một phiên kết nối đến MySQL. Áp dụng mẫu Singleton để tối ưu tài nguyên.

package utils;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class DatabaseHelper {
    private static final String URL = "jdbc:mysql://localhost:3306/quanlythuvien";
    private static final String USER = "root";
    private static final String PASS = "password123"; // Đổi thành mật khẩu của bạn
    
    private static Connection connection;

    public static Connection getConnection() {
        try {
            if (connection == null || connection.isClosed()) {
                // Tải driver MySQL
                Class.forName("com.mysql.cj.jdbc.Driver");
                connection = DriverManager.getConnection(URL, USER, PASS);
            }
        } catch (ClassNotFoundException | SQLException e) {
            System.err.println("Lỗi kết nối CSDL: " + e.getMessage());
        }
        return connection;
    }
}

III. Mô Hình Dữ Liệu (Model)

Khai báo các thuộc tính tương ứng với các cột trong cơ sở dữ liệu. Cần có Constructor, Getter và Setter.

package models;

public class Sach {
    private String maSach;
    private String tenSach;
    private String nhaXuatBan;
    private int soLuong;

    // Constructor
    public Sach(String maSach, String tenSach, String nhaXuatBan, int soLuong) {
        this.maSach = maSach;
        this.tenSach = tenSach;
        this.nhaXuatBan = nhaXuatBan;
        this.soLuong = soLuong;
    }

    // Getters and Setters
    public String getMaSach() { return maSach; }
    public void setMaSach(String maSach) { this.maSach = maSach; }
    
    public String getTenSach() { return tenSach; }
    public void setTenSach(String tenSach) { this.tenSach = tenSach; }
    
    public String getNhaXuatBan() { return nhaXuatBan; }
    public void setNhaXuatBan(String nhaXuatBan) { this.nhaXuatBan = nhaXuatBan; }
    
    public int getSoLuong() { return soLuong; }
    public void setSoLuong(int soLuong) { this.soLuong = soLuong; }
}

IV. Xây Dựng Data Access Object (DAO)

Nơi chứa toàn bộ câu lệnh SQL (INSERT, UPDATE, DELETE, SELECT). Lớp Controller sẽ gọi các hàm trong lớp này.

package dao;

import models.Sach;
import utils.DatabaseHelper;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.List;

public class SachDAO {
    
    // Thêm mới một sách vào CSDL
    public boolean themSach(Sach sach) {
        String sql = "INSERT INTO sach (ma_sach, ten_sach, nha_xuat_ban, so_luong) VALUES (?, ?, ?, ?)";
        
        try (Connection conn = DatabaseHelper.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setString(1, sach.getMaSach());
            pstmt.setString(2, sach.getTenSach());
            pstmt.setString(3, sach.getNhaXuatBan());
            pstmt.setInt(4, sach.getSoLuong());
            
            return pstmt.executeUpdate() > 0;
            
        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }

    // Lấy toàn bộ danh sách sách từ CSDL
    public List layDanhSachSach() {
        List list = new ArrayList<>();
        String sql = "SELECT * FROM sach";
        
        try (Connection conn = DatabaseHelper.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql);
             ResultSet rs = pstmt.executeQuery()) {
            
            while (rs.next()) {
                Sach s = new Sach(
                    rs.getString("ma_sach"),
                    rs.getString("ten_sach"),
                    rs.getString("nha_xuat_ban"),
                    rs.getInt("so_luong")
                );
                list.add(s);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return list;
    }
}

V. Controller & Tương Tác JavaFX

Lớp này liên kết với giao diện `.fxml`. Nó lấy dữ liệu người dùng nhập, tạo đối tượng Model, và chuyển cho DAO xử lý.

package controllers;

import dao.SachDAO;
import models.Sach;
import javafx.fxml.FXML;
import javafx.scene.control.Alert;
import javafx.scene.control.TextField;

public class ThemSachController {

    // Ánh xạ các UI Component từ file FXML
    @FXML private TextField txtMaSach;
    @FXML private TextField txtTenSach;
    @FXML private TextField txtNhaXuatBan;
    @FXML private TextField txtSoLuong;

    private SachDAO sachDAO;

    // Phương thức chạy ngay khi giao diện được load
    @FXML
    public void initialize() {
        sachDAO = new SachDAO();
    }

    // Phương thức được gắn vào sự kiện OnAction của nút "Lưu"
    @FXML
    public void xuLyLuuSach() {
        try {
            // 1. Lấy dữ liệu
            String ma = txtMaSach.getText();
            String ten = txtTenSach.getText();
            String nxb = txtNhaXuatBan.getText();
            int soLuong = Integer.parseInt(txtSoLuong.getText()); // Có thể ném lỗi NumberFormatException

            // 2. Kiểm tra rỗng (Validation cơ bản)
            if(ma.isEmpty() || ten.isEmpty()) {
                hienThiThongBao("Lỗi", "Mã và Tên sách không được để trống!");
                return;
            }

            // 3. Đóng gói vào Model
            Sach sachMoi = new Sach(ma, ten, nxb, soLuong);

            // 4. Gọi DAO thực thi
            boolean thanhCong = sachDAO.themSach(sachMoi);

            if (thanhCong) {
                hienThiThongBao("Thành công", "Đã lưu sách vào cơ sở dữ liệu!");
                xoaTrangForm();
            } else {
                hienThiThongBao("Thất bại", "Trùng mã sách hoặc lỗi CSDL.");
            }

        } catch (NumberFormatException e) {
            hienThiThongBao("Lỗi nhập liệu", "Số lượng phải là một số nguyên hợp lệ.");
        }
    }

    private void hienThiThongBao(String title, String content) {
        Alert alert = new Alert(Alert.AlertType.INFORMATION);
        alert.setTitle(title);
        alert.setHeaderText(null);
        alert.setContentText(content);
        alert.showAndWait();
    }

    private void xoaTrangForm() {
        txtMaSach.clear();
        txtTenSach.clear();
        txtNhaXuatBan.clear();
        txtSoLuong.clear();
    }
}
